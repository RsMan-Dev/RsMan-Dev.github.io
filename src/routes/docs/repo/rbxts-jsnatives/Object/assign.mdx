will assign all properties of the next objects to the first one, overriding existing properties from the first objects

```ts
const a = {a: 1};
const b = [1, 2, 3];
const c = new Set([1, 2, 3]);
const d = new Map([["a", 8], ["b", 2], ["c", 3]]);

print(Object.assign(a, b, c, d)); // {a: 8, 0: 1, 1: true, 2: true, 3: true, b: 2, c: 3}
```

<br/>

Works with proxies

```ts
const a = {c: 1}

const target = {}

const b = new Proxy(target, {
  set: (target, prop, value) => {
    target[prop] = value;
    return true;
  }
})

const c = new Proxy({}, {
  get: (target, prop) => {
    return prop === "a" ? 2 : prop;
  },
  ownKeys: () => ["a", "b"]
})

// remember that a proxy is technically an empty object
print(Object.assign(b, c, a), target, JSON.stringify(b)); // {} {a: 2, b: 2, c: 1}, {"a":2,"b":2,"c":1}
```
