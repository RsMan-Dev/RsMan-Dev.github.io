get all values of an object

const a = {a: 1, b: 2, c: 3};


print(Object.values(a)); // [1, 2, 3]

On proxies, Object.values will trigger the get trap to get the values

const a = new Proxy({}, {
  ownKeys: () => {
    return ["a", "b", "c"];
  },
  get: (target, prop) => {
    return prop;
  }
});


print(Object.values(a)); // ["a", "b", "c"]

Note that if the proxy returns undefined, the table may have different length than Object.keys

const a = new Proxy({}, {
  ownKeys: () => {
    return ["a", "b", "c"];
  },
  get: (target, prop) => {
    return prop === "b" ? undefined : prop;
  }
});


print(Object.values(a), Object.keys(a)); // ["a", "c"] ["a", "b", "c"]

const a = new Set([1, 2, 3]);


print(Object.values(a)); // [true, true, true]

const a = new Map([["a", 1], ["b", 2], ["c", 3]]);


print(Object.values(a)); // [1, 2, 3]

const a = [1, 2, 3];


print(Object.values(a)); // [1, 2, 3]
